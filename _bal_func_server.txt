// === BALANCOVÁNÍ BATERIE ===
// Pylontech 1×/30 dní — nabít na 100% SOC, pak držet do vybalancování
// Balanced = SOC>=100 + |I|<0.5A + cell ΔV<0.02V po 20 min nepřetržitě
// Priority: NIBE > balancování > patrony/auto/bazén

var patBlockCharge = global.get("pat_block_charge") || false;
var config = global.get("fve_config") || {};
var status = global.get("fve_status") || {};
var plan = global.get("fve_plan") || {};
var minSoc = config.min_soc || 20;
var currentSoc = status.battery_soc || 50;
var maxGridW = config.max_spotreba_sit_w || 22000;
var SAFETY_MARGIN = config.safety_margin_w || 2000;
var cerpadloTopi = msg.cerpadloTopi || false;
var autoCharging = msg.autoNabijeniAktivni || false;
var saunaAktivni = msg.saunaAktivni || false;

// === BATTERY SENSORS ===
var ha = global.get("homeassistant.homeAssistant.states") || {};
var batCurrent = parseFloat((ha["sensor.battery_dc_current"] || {}).state);
var maxCellV = parseFloat((ha["sensor.battery_max_cell_voltage"] || {}).state);
var minCellV = parseFloat((ha["sensor.battery_min_cell_voltage"] || {}).state);
var batSocPrecise = parseFloat((ha["sensor.battery_soc_precise"] || {}).state);
if (isNaN(batSocPrecise)) batSocPrecise = currentSoc;

// === INDOOR TEMP (pump blocking) ===
var indoorTemp = parseFloat((ha["sensor.hp2551ae_pro_v2_1_0_indoor_temperature"] || {}).state) || 22;
var targetTemp = parseFloat((ha["input_number.nastavena_teplota_v_dome"] || {}).state) || 23;

// === PRICE & SOLAR CONTEXT ===
var prices = global.get("fve_prices_forecast") || [];
var hour = new Date().getHours();
var PRAH_LEVNA = config.prah_levna_energie || 4;
var lvl = 99;
for (var i = 0; i < prices.length; i++) {
    if (prices[i].hour === hour && (prices[i].minute === 0 || prices[i].minute === undefined)) {
        lvl = prices[i].levelCheapestHourBuy || 99;
        break;
    }
}
var solarStart = (plan && plan.config && plan.config.solarStartHour) || 9;
var solarEnd = (plan && plan.config && plan.config.solarEndHour) || 17;
var isSolarHour = (hour >= solarStart && hour < solarEnd);

var pvEntity = ha["sensor.vyroba_fve"] || {};
var currentSolar = Math.round(parseFloat(pvEntity.state) || 0);
var solarPassthrough = currentSolar > 10 ? Math.max(50, currentSolar) : 0;
var gridBias = config.setrit_grid_bias_w || 150;


// === SELL PRICE (negative = zákaz přetoků) ===
var currentSellPrice = 99;
for (var pi = 0; pi < prices.length; pi++) {
    if (prices[pi].hour === hour && (prices[pi].minute === 0 || prices[pi].minute === undefined)) {
        currentSellPrice = parseFloat(prices[pi].priceCZKhourProd) || 99;
        break;
    }
}
var negativeSellPrice = currentSellPrice <= 0;
// === VICTRON SETTINGS ===
var shouldCharge = isSolarHour || lvl <= PRAH_LEVNA;
var consumers = [];
if (cerpadloTopi) consumers.push("Topeni");
if (autoCharging) consumers.push("Auto");
if (saunaAktivni) consumers.push("Sauna");

var psp;
if (shouldCharge) {
    // Solar or cheap hour → charge aggressively
    if (cerpadloTopi || autoCharging) {
        psp = maxGridW - SAFETY_MARGIN;
    } else {
        psp = -maxGridW;
    }
    msg.victron = {
        power_set_point: psp,
        min_soc: minSoc,
        schedule_soc: 100,
        schedule_charge_duration: 86399,
        schedule_charge_day: 7,
        max_discharge_power: -1,
        max_charge_power: patBlockCharge ? 0 : -1,
        feedin_on: !negativeSellPrice,
        max_feed_in_power: negativeSellPrice ? 0 : (config.max_feed_in_w || 7600),
        prevent_feedback: negativeSellPrice ? 1 : 0
    };
} else {
    // Expensive hour → hold battery (don't charge from grid, don't discharge)
    psp = gridBias;
    msg.victron = {
        power_set_point: gridBias,
        min_soc: minSoc,
        schedule_soc: 0,
        schedule_charge_duration: 0,
        schedule_charge_day: -7,
        max_discharge_power: solarPassthrough,
        max_charge_power: patBlockCharge ? 0 : -1,
        feedin_on: !negativeSellPrice,
        max_feed_in_power: negativeSellPrice ? 0 : (config.max_feed_in_w || 7600),
        prevent_feedback: negativeSellPrice ? 1 : 0
    };
}
msg.mode = "Balancování";

// === PUMP BLOCKING ===
var blockPump = indoorTemp >= (targetTemp - 0.2);
global.set("balancing_block_pump", blockPump);

// === SOLAR DUMP (záporná cena + SOC blízko 100% = signál pro patrony/auto) ===
var needSolarDump = negativeSellPrice && batSocPrecise >= 95 && currentSolar > 500;
global.set("balancing_solar_dump", needSolarDump);
if (needSolarDump) {
    node.warn("BALANCOVANI: Solar dump aktivní (sell=" + currentSellPrice.toFixed(2) + ", SOC=" + batSocPrecise + "%, solar=" + currentSolar + "W)");
}
// === BALANCED DETECTION ===
var STABLE_DURATION_MS = 20 * 60 * 1000;
var MAX_CURRENT_A = 0.5;
var MAX_CELL_DIFF_V = 0.02;

var socOk = batSocPrecise >= 100;
var currentOk = !isNaN(batCurrent) && Math.abs(batCurrent) < MAX_CURRENT_A;
var cellDiff = (!isNaN(maxCellV) && !isNaN(minCellV)) ? Math.abs(maxCellV - minCellV) : -1;
var cellOk = cellDiff < 0 || cellDiff < MAX_CELL_DIFF_V;

var stableSince = flow.get("balancing_stable_since") || 0;
var balanced = false;
var detectionStatus = "";

if (socOk && currentOk && cellOk) {
    if (stableSince === 0) {
        stableSince = Date.now();
        flow.set("balancing_stable_since", stableSince);
        detectionStatus = "Stabilizace začala";
    } else {
        var elapsed = Date.now() - stableSince;
        if (elapsed >= STABLE_DURATION_MS) {
            balanced = true;
            detectionStatus = "VYBALANCOVÁNO!";
        } else {
            detectionStatus = "Stabilizace " + Math.round(elapsed / 60000) + "/" + Math.round(STABLE_DURATION_MS / 60000) + "min";
        }
    }
} else {
    stableSince = 0;
    flow.set("balancing_stable_since", 0);
    var reasons = [];
    if (!socOk) reasons.push("SOC=" + batSocPrecise + "%");
    if (!currentOk) reasons.push("I=" + (isNaN(batCurrent) ? "N/A" : batCurrent.toFixed(2)) + "A");
    if (cellDiff >= 0 && !cellOk) reasons.push("\u0394V=" + cellDiff.toFixed(3) + "V");
    detectionStatus = reasons.join(", ");
}

// === CONFIGURABLE FORCE STOP at SOC=100% ===
var forceStopHours = parseFloat(config.balancing_force_stop_hours) || 2;
var MAX_SOC100_MS = forceStopHours * 60 * 60 * 1000;
var soc100since = flow.get("balancing_soc100_since") || 0;
if (batSocPrecise >= 100) {
    if (soc100since === 0) {
        soc100since = Date.now();
        flow.set("balancing_soc100_since", soc100since);
    } else if (!balanced && (Date.now() - soc100since) >= MAX_SOC100_MS) {
        balanced = true;
        detectionStatus = "FORCE STOP (1h@100%)";
        node.warn("=== BATTERY BALANCING FORCE STOP (1h at 100% SOC) ===");
    }
} else {
    flow.set("balancing_soc100_since", 0);
}

// v23: Orchestrator force-stop — route through here so service call fires
if (msg.forceStopBalancing && !global.get("balancing_completed")) {
    balanced = true;
    var isNaturalBalanced = false;
    detectionStatus = "FORCE STOP (orchestrator max)";
    node.warn("=== BATTERY BALANCING FORCE STOP (orchestrator) ===");
}

msg.balancing = {
    batCurrent: isNaN(batCurrent) ? null : Math.round(batCurrent * 100) / 100,
    cellDiff: cellDiff >= 0 ? Math.round(cellDiff * 1000) / 1000 : null,
    socPrecise: batSocPrecise,
    detectionStatus: detectionStatus
};

// === STATUS ===
var statusText = (shouldCharge ? "NAB\u00cdJ\u00cdM" : "DR\u017e\u00cdM") + " | SOC:" + currentSoc + "% | " + detectionStatus;
if (blockPump) statusText += " | PUMP_BLK";
node.status({fill: balanced ? "green" : "cyan", shape: "dot", text: statusText});

node.warn("BALANCOVANI: " + statusText + " | psp=" + psp + " sched_soc=" + (shouldCharge ? 100 : 0));

global.set("energy_arbiter", {
    mode: "Balancování",
    battery_charging: shouldCharge,
    battery_charge_rate_w: shouldCharge ? 5000 : 0,
    power_set_point: psp,
    consumers_active: consumers,
    max_discharge_allowed: false,
    timestamp: Date.now()
});

// Output 1: msg → Victron + Log (always)
// Output 2: HA service call (only when balanced)
var svcMsg = null;
var boolMsg = null;
if (balanced && !global.get("balancing_completed")) {
    flow.set("balancing_stable_since", 0);
    flow.set("balancing_soc100_since", 0);
    global.set("balancing_completed", true);
    global.set("balancing_started_at", null);
    var isForceStop = detectionStatus.indexOf("FORCE STOP") >= 0;
    node.warn("=== BATTERY BALANCING " + (isForceStop ? "FORCE STOP" : "COMPLETE") + "! ===");
    svcMsg = {
        payload: {
            data: {
                datetime: new Date().toISOString().replace("T", " ").substring(0, 19)
            }
        }
    };
    boolMsg = {
        payload: {
            action: "input_boolean." + (isForceStop ? "turn_off" : "turn_on"),
            target: { entity_id: ["input_boolean.pylontech_balancing_ok"] },
            data: {}
        }
    };
}

return [msg, svcMsg, boolMsg];

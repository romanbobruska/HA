// FVE Heating Control v2.5 — Kompletní řízení topení domu
// Řídí: NIBE (47371), oběhové čerpadlo, patrony (3 fáze), bazénový ventil, chlazení

var config = global.get("fve_config") || {};
var prices = global.get("fve_prices_forecast") || [];

// === HELPER ===
function getState(id) {
    var e = global.get("homeassistant.homeAssistant.states['" + id + "']") || {};
    return e.state;
}
function getFloat(id, def) {
    var v = parseFloat(getState(id));
    return isNaN(v) ? (def || 0) : v;
}
function getBool(id) { return getState(id) === "on"; }

// Najdi nejlevnější budoucí hodinu (do konce solární výroby nebo do rána)
function hasCheaperHourAhead(prices, currentHour, currentLevel) {
    for (var i = 0; i < prices.length; i++) {
        var h = prices[i].hour;
        var m = prices[i].minute || 0;
        if (m !== 0) continue;
        // Budoucí hodiny (po aktuální)
        if (h <= currentHour) continue;
        var futureLevel = prices[i].levelCheapestHourBuy || 99;
        if (futureLevel < currentLevel) return true;
    }
    return false;
}

// === PARAMETRY Z CONFIGU ===
var MIN_TANK = config.topeni_min_teplota_nadrze || 30;
var MAX_TANK = config.topeni_max_teplota_nadrze || 50;
var NOCNI_SNIZ = config.topeni_nocni_snizeni || 0.5;
var NOCNI_OD = config.topeni_nocni_od || 22;
var NOCNI_DO = config.topeni_nocni_do || 6;
var NOUZ_TEMP = config.topeni_nouzova_teplota || 18;
var MIN_PRETOK = config.topeni_min_pretok_patron_w || 3000;
var PATRON_W = config.topeni_patron_faze_w || 3000;
var MIN_SOC_PAT = config.topeni_min_soc_patron || 90;
var MAX_TANK_PAT = getFloat("input_number.topeni_max_teplota_patron", config.topeni_max_teplota_patron || 53);
var PRAH_DRAHA = config.prah_draha_energie || 12;
var BEZPECNY_POKLES = config.topeni_bezpecny_pokles || 0.7;
var HYSTEREZE_CHLAZENI = config.topeni_hystereze_chlazeni || 1.0;
var SOLAR_FORECAST_THRESHOLD = config.topeni_solar_forecast_kwh || 30;
// v19.5: Solar forecast source z configu (VICTRON / OPEN_METEO)
var solarSource = (config.solar_forecast_source || "VICTRON").toUpperCase();
var entityZitra = (solarSource === "OPEN_METEO") ? "sensor.energy_production_tomorrow_3" : "input_number.predpoved_solarni_vyroby_zitra";
var solarForecastTomorrow = getFloat(entityZitra, 0);
var bigSolarTomorrow = solarForecastTomorrow >= SOLAR_FORECAST_THRESHOLD;
var letniRezim = config.letni_rezim === true;

// === ČTENÍ STAVŮ ===
var automatizace = getBool("input_boolean.automatizovat_topeni");
var targetTemp = getFloat("input_number.nastavena_teplota_v_dome", 22);
var indoorTemp = getFloat("sensor.hp2551ae_pro_v2_1_0_indoor_temperature", 20);
var outdoorTemp = getFloat("sensor.hp2551ae_pro_v2_1_0_outdoor_temperature", 10);
var tankTemp = getFloat("sensor.teplota_smesovac_teplota", 35);
var dm = getFloat("sensor.nibe_degree_minutes", 0);
var batSoc = getFloat("sensor.battery_percent", 50);
var prebytek = getFloat("sensor.fve_dostupny_prebytek", 0);
var compressor = getBool("binary_sensor.nibe_kompresory_aktivni_binarni");
var pumpState = getState("sensor.nibe_aktualni_realny_stav") || "";
var fireplace = getBool("input_boolean.topi_se_v_krbu");
var coolEnabled = getBool("input_boolean.chlazeni");
var autoHlad = getBool("input_boolean.auto_ma_hlad");
var autoNabiji = global.get("auto_nabijeni_aktivni") || false;  // wallbox fyzicky nabiji
// v24: ultraLevna se počítá níže (po isSolarHour)
var ultraLevna = false;
// v22: Čteme z HA entity (přežije restart NR) i z globalu
var _fvePlanAttrs = (global.get("homeassistant.homeAssistant.states['sensor.fve_plan']") || {}).attributes || {};
var balancingActive = _fvePlanAttrs.current_mode === "Balancování" || global.get("balancing_active") || false;
var balancingSolarDump = global.get("balancing_solar_dump") || false;  // Pylontech balancing mode
var balancingBlockPump = global.get("balancing_block_pump") || false;  // block pump during balancing

var nibeOn = getBool("switch.nibe_topeni");
var obehOn = getBool("switch.teplota_smesovac");
var p1on = getBool("switch.patrona_faze_1");
var p2on = getBool("switch.patrona_faze_3_2");
var p3on = getBool("switch.patrona_faze_3");
var ventilOn = getBool("switch.bazen_ventil_smesovac");
var coolOn = getBool("switch.nibe_chlazeni");

var hour = new Date().getHours();
var now = Date.now();
var NIBE_COOLDOWN_MS = (config.nibe_cooldown_min || 10) * 60 * 1000;  // min doba mezi NIBE přepnutími
var lastNibeSwitch = flow.get("last_nibe_switch") || 0;
var nibeCooldownOk = (now - lastNibeSwitch) >= NIBE_COOLDOWN_MS;

// === GRID LOST ===
var gridLostRaw = getFloat("sensor.grid_lost_ciselny", 0);
var gridLost = gridLostRaw >= 2;  // 0=OK, 1=Warning, 2=Grid Lost
var solarPower = getFloat("sensor.vyroba_fve", 0);  // aktuální solární výkon (W)
var NIBE_PEAK_W = config.nibe_peak_w || 14000;  // peak odběr NIBE

// === SOLÁRNÍ HODINY ===
var plan = global.get("fve_plan") || {};
var solarStart = (plan && plan.solarStartHour) || 9;
var solarEnd = (plan && plan.solarEndHour) || 17;
var isSolarHour = (hour >= solarStart && hour < solarEnd);
var solarHoursLeft = isSolarHour ? (solarEnd - hour) : 0;
// v24: Výpočet ultraLevna přímo z cen — nezávislý na globálu z mode flow
var PRAH_ULTRA_LEVNA_HTG = config.prah_ultra_levna_nakup || 1.0;
var _ultraBuyPrice = 99;
var _ultraSellPrice = 99;
for (var _ui = 0; _ui < prices.length; _ui++) {
    if (prices[_ui].hour === hour && (prices[_ui].minute === 0 || prices[_ui].minute === undefined)) {
        _ultraBuyPrice = parseFloat(prices[_ui].priceCZKhourBuy) || 99;
        _ultraSellPrice = parseFloat(prices[_ui].priceCZKhourProd) || 99;
        break;
    }
}
ultraLevna = isSolarHour && _ultraBuyPrice < PRAH_ULTRA_LEVNA_HTG && _ultraSellPrice <= 0;
var entityZbyvajici = (solarSource === "OPEN_METEO") ? "sensor.energy_production_today_remaining_3" : "input_number.zbyvajici_solarni_vyroba_dnes";
var zbyvajiciSolar = getFloat(entityZbyvajici, 0) * 1000; // Wh
var kapacitaBat = (config.kapacita_baterie_kwh || 28) * 1000; // Wh
var SOLAR_MARGIN_HOURS = config.topeni_solar_margin_hours || 2;

// === CENOVÝ LEVEL (vždy celá hodina, minute=0) ===
var lvl = 99;
for (var i = 0; i < prices.length; i++) {
    if (prices[i].hour === hour && (prices[i].minute === 0 || prices[i].minute === undefined)) {
        lvl = prices[i].levelCheapestHourBuy || 99;
        break;
    }
}
// Fallback: pokud se aktuální hodina nenašla v prices (po restartu NR nebo midnight edge case)
if (lvl === 99) {
    lvl = getFloat("input_number.levelcheapesthourbuy", 12);
}
// v19.3: Plan setrit = baterie se nevybiji, NIBE muze topit ze site
var planCurrentMode = (plan && plan.currentMode) || "normal";
var isDraha = (lvl >= PRAH_DRAHA) && (planCurrentMode !== "setrit");

// === NOČNÍ REŽIM ===
var isNight = (hour >= NOCNI_OD || hour < NOCNI_DO);
// Noční snížení = hystereze pro drahé hodiny (v levných topit na plnou teplotu)
var effTarget = isNight ? (targetTemp - NOCNI_SNIZ) : targetTemp;

// === PATRONY STATE ===
var actPat = (p1on ? 1 : 0) + (p2on ? 1 : 0) + (p3on ? 1 : 0);
var availPat = prebytek + actPat * PATRON_W;

// === OCHRANA ===
var pumpWorking = pumpState.indexOf("Klidový") === -1 && pumpState !== "";
var canOffNibe = !compressor && !pumpWorking;
var canOnNibe = nibeCooldownOk;

// === ROZHODOVÁNÍ ===
var actions = [];
var reason = "";
var patTarget = 0;
var shutdownDone = flow.get("shutdown_done") || false;

if (!automatizace) {
    // AUTOMATIZACE OFF — manuální mód dle input_select.topeni_mod
    // Flow NEPŘEPISUJE mod — jen provádí příkazy dle manuálně nastavenémo modu
    var manualMod = getState("input_select.topeni_mod") || "Vypnuto";
    flow.set("topeni_mod_active", manualMod);  // sync flow state s HA entitou

    if (manualMod === "Vypnuto") {
        // Manuální VYPNUTO — žádné zásahy, uživatel ovládá vše sám
        reason = "Manuální mod: VYPNUTO — bez zásahů automatizace";

    } else if (manualMod === "NIBE") {
        // Manualní NIBE – uživatel řídí NIBE přímo, automatizace NEZASAHUJE do NIBE
        // Pouze patrony zablokujeme (bezpečnost)
        if (p1on) actions.push("p1_off");
        if (p2on) actions.push("p2_off");
        if (p3on) actions.push("p3_off");
        reason = "Manuální mod: NIBE – bez zásahů do NIBE | indoor=" + indoorTemp.toFixed(1) + "°C";

    } else if (manualMod === "Patrony") {
        // Manuální PATRONY — topit jen z patron, NIBE blokováno
        // NIBE: nezasahovat – uživatel kontroluje
        var tankOk = tankTemp >= MIN_TANK;
        var patronyMohou = batSoc >= MIN_SOC_PAT && !autoHlad && !autoNabiji && tankTemp < MAX_TANK_PAT && (!gridLost || solarPower > 0);
        // Hlavní loop zapíná patrony POUZE pokud actPat === 0.
        // Pokud patrony běží (actPat > 0), 5s smyčka řídí počet fází.
        if (patronyMohou && actPat === 0) {
            if (availPat >= MIN_PRETOK * 3)      { patTarget = 3; }
            else if (availPat >= MIN_PRETOK * 2) { patTarget = 2; }
            else if (availPat >= MIN_PRETOK)     { patTarget = 1; }
        } else if (patronyMohou && actPat > 0) {
            // Patrony běží — 5s smyčka řídí
            patTarget = actPat;
        }
        // patronyMohou === false → hlavní loop vypne fáze
        var p1want = patTarget >= 1;
        var p2want = patTarget >= 2;
        var p3want = patTarget >= 3;
        if (p1want !== p1on) actions.push(p1want ? "p1_on" : "p1_off");
        if (p2want !== p2on) actions.push(p2want ? "p2_on" : "p2_off");
        if (p3want !== p3on) actions.push(p3want ? "p3_on" : "p3_off");
        reason = "Manuální mod: PATRONY | přebytek=" + Math.round(availPat) + "W fáze=" + patTarget;

    } else if (manualMod === "Obehove") {
        // Manualní OBEHOVE – oběhové čerpadlo, NIBE nezasahovat
        // Patrony zastavit
        if (p1on) actions.push("p1_off");
        if (p2on) actions.push("p2_off");
        if (p3on) actions.push("p3_off");
        var tankOk = tankTemp >= MIN_TANK;
        var needsHeat = indoorTemp < effTarget;
        if (tankOk && needsHeat && !obehOn) {
            actions.push("obehove_on");
            reason = "Manuální mod: OBĚHOVÉ — ON";
        } else if ((!tankOk || !needsHeat) && obehOn) {
            actions.push("obehove_off");
            reason = "Manuální mod: OBĚHOVÉ — OFF";
        } else {
            reason = "Manuální mod: OBĚHOVÉ | nádrž=" + tankTemp.toFixed(1) + "°C";
        }
    }

    // Zapiš aktuální manuální mod do global (pro HA input_select update)
    global.set("topeni_mod", manualMod);

} else {
    var forceStart = flow.get("shutdown_done") === true;  // právě zapnuta automatizace
    flow.set("shutdown_done", false);
    // Reset summer flag pokud se změní automatizace
    if (letniRezim) {
        // LETNÍ REŽIM — jednorázové nastavení default stavu, pak jen chlazení
        var summerDone = flow.get("summer_setup_done") || false;
        if (!summerDone) {
            // Jednorázově: NIBE on (pro TUV), oběhové off, patrony off
            if (canOffNibe) actions.push("nibe_on");
            if (obehOn) actions.push("obehove_off");
            if (p1on) actions.push("p1_off");
            if (p2on) actions.push("p2_off");
            if (p3on) actions.push("p3_off");
            flow.set("summer_setup_done", true);
            reason = "Léto — jednorázové nastavení default";
        }

        // Chlazení — běží stále v létě
        if (coolEnabled) {
            if (!coolOn && indoorTemp > effTarget + HYSTEREZE_CHLAZENI) {
                actions.push("cool_on");
                reason = "Chlazení ON (" + indoorTemp.toFixed(1) + ">" + (effTarget+HYSTEREZE_CHLAZENI).toFixed(1) + "°C)";
            } else if (coolOn && indoorTemp < effTarget) {
                actions.push("cool_off");
                reason = "Chlazení OFF (" + indoorTemp.toFixed(1) + "<" + effTarget.toFixed(1) + "°C)";
            } else {
                reason = "Léto | " + indoorTemp.toFixed(1) + "/" + effTarget.toFixed(1) + "°C" + (coolOn ? " | Chlazení" : "");
            }
        } else {
            if (coolOn) actions.push("cool_off");
            reason = "Léto | chlazení vypnuto";
        }
    } else {
        // ZIMNÍ REŽIM
        flow.set("summer_setup_done", false);

        // 1. Bazénový ventil - kontrola
        if (ventilOn) actions.push("ventil_off");

        // 2. Chlazení OFF v zimě
        if (coolOn) actions.push("cool_off");

        // 3. TOPÉNÍ MOD — vzájemná exkluzivita NIBE a patrony
        // MOD_PATRONY → NIBE blokováno. MOD_NIBE → patrony blokovány.
        var topeniModAktivni = flow.get("topeni_mod_active") || "Vypnuto";

        // Podmínky pro topení
        var tankOk = tankTemp >= MIN_TANK;
        var needsHeat = indoorTemp < effTarget;

        // Podmínky pro spuštění patrony (solární přebytek)
        // Ultra levná: relaxace - nižší SOC práh, patrony mohou brát ze sítě, auto neblokuje
        var ultraSocPrah = ultraLevna ? (config.min_soc || 20) : MIN_SOC_PAT;
        // v19.10: Balancing = patrony blokovány (priority: balancing > patrony)
        var patronyMohou = (!balancingActive || balancingSolarDump) && batSoc >= ultraSocPrah && (ultraLevna || balancingSolarDump || (!autoHlad && !autoNabiji)) && tankTemp < MAX_TANK_PAT && (!gridLost || solarPower > 0);
        var patronyActive = false;
        // Hlavní loop zapíná patrony POUZE pokud actPat === 0 (první zapnutí nebo restart).
        // Jakmile patrony jednou běží, 5s smyčka (pat_korekce_func) přebírá řízení počtu fází.
        if (patronyMohou && actPat === 0) {
            // Patrony neběží — konzervativní start: vždy jen 1 fáze, korekce (5s) přidá další
            // Ultra levná: patrony mohou startovat i bez solárního přebytku (grid je levný)
            if (ultraLevna || availPat >= MIN_PRETOK) {
                patTarget = 1; patronyActive = true;
                // Nastav cooldown pro korekci — čekat než se systém stabilizuje po startu
                flow.set("pat_change_cooldown", 6); // 6×5s = 30s
            }
        } else if (patronyMohou && actPat > 0) {
            // Patrony běží — 5s smyčka řídí, hlavní loop jen zaznamená stav
            patTarget = actPat;
            patronyActive = true;
        }
        // patronyMohou === false → hlavní loop vypne fáze (podmínky nesplněny)

        // Přepínání modu — patrony POUZE pokud teplota je blízko cíle + reálná šance na běh
        // Patrony = POSLEDNÍ priorita — přebytky co nemám kam dát (auto, NIBE, baterie)
        // Velký rozdíl teploty → NIBE (topení domu). Malý rozdíl + reálné patrony → patrony.
        var PATRON_TEMP_MARGIN = 0.3;  // °C — max rozdíl indoor vs target pro patrony
        var tempGap = effTarget - indoorTemp;  // kladné = pod cílem
        // Kontrola reálnosti patrony:
        // 1. SOC musí být blízko MIN_SOC_PAT (max 15% pod) — jinak NIBE topí v levných hodinách
        // 2. Zbývající solar musí pokrýt nabití baterie + spotřebu domu s 50% rezervou
        var patronySocReady = batSoc >= (MIN_SOC_PAT - 15);
        var battNeedWh = Math.max(0, (MIN_SOC_PAT - batSoc) / 100 * kapacitaBat);
        var consumeEstWh = solarHoursLeft * 1500; // ~1.5kWh/h průměrná spotřeba domu
        var patronyRealisticke = ultraLevna || (patronySocReady && zbyvajiciSolar > (battNeedWh + consumeEstWh) * 1.5);
        if (isSolarHour && tempGap <= PATRON_TEMP_MARGIN && needsHeat && patronyRealisticke) {
            // Solární hodiny + teplota blízko cíle + reálná šance na SOC ≥ 95% → mod Patrony
            topeniModAktivni = "Patrony";
        } else if (tempGap > PATRON_TEMP_MARGIN && needsHeat) {
            // Velký rozdíl — dům potřebuje topit přes NIBE
            topeniModAktivni = "NIBE";
        } else if (!needsHeat && patronyActive) {
            // Teplota dosažena + patrony běží → patrony dál ohřívají nádrž
            topeniModAktivni = "Patrony";
        } else if (!needsHeat) {
            topeniModAktivni = "Vypnuto";
        } else {
            // Fallback: needsHeat + ne solární hodiny → NIBE
            topeniModAktivni = "NIBE";
        }
        flow.set("topeni_mod_active", topeniModAktivni);

        // v24 DEBUG: Diagnostika patrony vs NIBE
        node.warn("HTG_DEBUG: ultraLevna=" + ultraLevna + " buyP=" + _ultraBuyPrice + " sellP=" + _ultraSellPrice + " patronyMohou=" + patronyMohou + " patronyActive=" + patronyActive + " needsHeat=" + needsHeat + " topeniMod=" + topeniModAktivni + " batSoc=" + batSoc + " ultraSocPrah=" + ultraSocPrah + " autoHlad=" + autoHlad + " patTarget=" + patTarget + " actions=" + actions.join(","));

        // Blokace dle modu
        var nibeBlockedByMod     = (topeniModAktivni === "Patrony");
        var patronyBlockedByMod  = (topeniModAktivni === "NIBE");

        // Patrony ON/OFF (blokované pokud MOD_NIBE)
        // Pokud patrony už běží (actPat > 0), fáze řídí korekce (5s) — hlavní loop jen startuje/stopuje
        var korekceRidi = actPat > 0 && patronyMohou;
        var p1want = !patronyBlockedByMod && patTarget >= 1;
        var p2want = !patronyBlockedByMod && patTarget >= 2;
        var p3want = !patronyBlockedByMod && patTarget >= 3;
        if (!korekceRidi) {
            if (p1want !== p1on) actions.push(p1want ? "p1_on" : "p1_off");
            if (p2want !== p2on) actions.push(p2want ? "p2_on" : "p2_off");
            if (p3want !== p3on) actions.push(p3want ? "p3_on" : "p3_off");
        }

        // 4. NIBE řízení v2.3 — NIBE topí NÁDRŽ, oběhové čerpadlo topí DŮM z nádrže
        //    NIBE se NESMÍ spustit/pokračovat v drahých hodinách (výjimka: velký solár)
        //    V levných hodinách: proaktivně topit nádrž i když dům je OK
        var safeTemp = effTarget - BEZPECNY_POKLES;
        // v19.4: Allow deferral to cheaper hour when temp is above safe minimum
        var canDeferHeat = indoorTemp >= safeTemp;
        var nibeBlockedByPatrony = nibeBlockedByMod;
        var nibeBlockDischarge = true;
        var gridAllowNibe = !gridLost || (gridLost && solarPower >= NIBE_PEAK_W);
        var cheaperAhead = hasCheaperHourAhead(prices, hour, lvl);
        var SOLAR_OVERRIDE_W = config.topeni_solar_override_w || 8000;
        // Proaktivní topení: v levné hodině topit nádrž i když dům nepotřebuje (nádrž < MAX_TANK)
        var tankNeedsHeat = tankTemp < MAX_TANK;
        var nibeWanted = needsHeat || (!isDraha && tankNeedsHeat);

        // === PRIORITA: TOPENÍ > AUTO NABÍJENÍ ===
        if (!ultraLevna && autoNabiji && nibeWanted && !nibeBlockedByPatrony) {
            actions.push("stop_auto_nabijeni");
        }

        // === BEZPEČNOSTNÍ KONTROLY (vždy první) ===
        if (!gridAllowNibe) {
            if (nibeOn && canOffNibe) actions.push("nibe_off");
        } else if (fireplace) {
            if (nibeOn && canOffNibe) actions.push("nibe_off");
        } else if (tankTemp > MAX_TANK) {
            if (nibeOn && canOffNibe) actions.push("nibe_off");
        } else if (nibeBlockedByPatrony) {
            if (nibeOn && canOffNibe) actions.push("nibe_off");

        // === v22: BALANCOVÁNÍ → NIBE OFF pokud dům nepotřebuje topit ===
        // Priorita: topení domu > balancování > proaktivní ohřev nádrže
        } else if (balancingActive && !needsHeat) {
            if (nibeOn && canOffNibe) actions.push("nibe_off");

        // === DRAHÁ HODINA → NIBE OFF (jediná výjimka: velký solár) ===
        } else if (isDraha) {
            if (prebytek >= SOLAR_OVERRIDE_W) {
                // Velký solární přebytek → NIBE ON/pokračovat (free energy)
                if (!nibeOn) actions.push("nibe_on");
                nibeBlockDischarge = false;
            } else {
                // Drahá hodina bez soláru → NIBE OFF, NESPOUŠTĚT
                if (nibeOn && canOffNibe) actions.push("nibe_off");
            }

        // === LEVNÁ/STŘEDNÍ HODINA ===
        } else if (nibeWanted) {
            // Levná hodina + (dům potřebuje topit NEBO nádrž pod MAX_TANK)
            // v19.3: Plan setrit = plan už rozhodl že hodina je levná → skip cheaperAhead/bigSolar pro nádrž
            // v19.4: Skip cheaperAhead ONLY when hour is genuinely cheap (lvl ≤ PRAH_LEVNA)
        // Prevents heating at level 5 when level 2 is next hour
        var PRAH_LEVNA = config.prah_levna_energie || 4;
        var planSaysChap = (planCurrentMode === "setrit") && (lvl <= PRAH_LEVNA);
            if (!planSaysChap && !isSolarHour && bigSolarTomorrow && (!needsHeat || canDeferHeat)) {
                // Velký solar zítra + dům nepotřebuje → šetříme pro solár
                if (nibeOn && canOffNibe) actions.push("nibe_off");
            } else if (!planSaysChap && !isSolarHour && cheaperAhead && (!needsHeat || canDeferHeat)) {
                // Levnější hodiny existují + dům nepotřebuje → počkat
                if (nibeOn && canOffNibe) actions.push("nibe_off");
            } else if (prebytek >= SOLAR_OVERRIDE_W) {
                // Velký solární přebytek → NIBE ON bez blokace vybíjení
                if (!nibeOn) actions.push("nibe_on");
                nibeBlockDischarge = false;
            } else {
                // Levná hodina → NIBE ON, topit nádrž
                if (!nibeOn) actions.push("nibe_on");
                nibeBlockDischarge = (batSoc <= 90);
            }

        // === DŮM NEPOTŘEBUJE + NÁDRŽ OK ===
        } else {
            if (nibeOn && canOffNibe) actions.push("nibe_off");
        }

        // Uložit info o blokaci vybíjení pro NIBE
        flow.set("nibe_block_discharge", nibeBlockDischarge);
        
        // 5. Oběhové čerpadlo (při krbu NESAHAT - řídí jiná automatika)
        // Oběhové je nezávislé na NIBE — čerpá teplo z nádrže do domu kdykoli nádrž má co dát
        // bigSolarTomorrow omezení se týká jen NIBE, ne oběhového (oběhové spotřebuje minimum)
        if (!fireplace) {
            // Noční snížení: oběhové nespouštět pokud teplota neklesla pod targetTemp - NOCNI_SNIZ
            var obehoveTarget = isNight ? (targetTemp - NOCNI_SNIZ) : effTarget;
            var needsHeatObehove = indoorTemp < obehoveTarget;
            // v19.10: Balancing = čerpadlo off (nevybíjet baterii), pokud teplota OK
            if (balancingBlockPump) {
                if (obehOn) actions.push("obehove_off");
            } else if (tankOk && needsHeatObehove) {
                actions.push("obehove_on");
            } else {
                // Nádrž vychladlá nebo teplota v noci OK → čerpadlo off
                if (obehOn) actions.push("obehove_off");
            }
        }

        // Reason
        var parts = [];
        parts.push(isDraha ? "DRAHÁ(Lv" + lvl + ")" : "Lv" + lvl);
        parts.push("In:" + indoorTemp.toFixed(1) + "/" + effTarget.toFixed(1) + "°C");
        parts.push("Tank:" + tankTemp.toFixed(0) + "°C");
        parts.push("DM:" + Math.round(dm));
        parts.push("Out:" + outdoorTemp.toFixed(0) + "°C");
        if (patronyActive) parts.push("PAT:" + patTarget + "f(" + Math.round(availPat) + "W)");
        if (fireplace) parts.push("KRB");
        if (isNight) parts.push("NOC");
        if (ultraLevna) parts.push("ULTRA_LEVNA");
        if (balancingActive) parts.push("⚡BAL");
        if (gridLost) parts.push("⚡GRID_LOST");
        if (isSolarHour) parts.push("☀" + solarHoursLeft + "h");
        if (cheaperAhead) parts.push("LEVNĚJŠÍ↓");
        if (bigSolarTomorrow) parts.push("☀Z:" + Math.round(solarForecastTomorrow) + "kWh");
        parts.push("Safe:" + safeTemp.toFixed(1) + "°C");
        if (!nibeBlockDischarge) parts.push("NoBLK");
        reason = parts.join(" | ");
    }
}

// === OCHRANA: NIBE nelze vypnout pokud pracuje ===
if (actions.indexOf("nibe_off") > -1 && !canOffNibe) {
    actions = actions.filter(function(a) { return a !== "nibe_off"; });
    reason += " ⚠️OCHRANA:NIBE pracuje";
}

// === OCHRANA: nikdy nibe_on + nibe_off zároveň ===
var hasNibeOn = actions.indexOf("nibe_on") > -1;
var hasNibeOff = actions.indexOf("nibe_off") > -1;
if (hasNibeOn && hasNibeOff) {
    // Konflikt — ponechat aktuální stav (nic nedělat)
    actions = actions.filter(function(a) { return a !== "nibe_on" && a !== "nibe_off"; });
    reason += " ⚠️KONFLIKT:on+off→skip";
}

// === COOLDOWN: min doba mezi NIBE přepnutími (přeskočit při forceStart) ===
if ((hasNibeOn && !nibeOn) || (hasNibeOff && nibeOn && canOffNibe)) {
    if (!nibeCooldownOk && !forceStart) {
        actions = actions.filter(function(a) { return a !== "nibe_on" && a !== "nibe_off"; });
        reason += " ⏳COOLDOWN";
    } else {
        flow.set("last_nibe_switch", now);
    }
}

// === FINÁLNÍ BEZPEČNOSTNÍ POJISTKA: NIBE a patrony nikdy současně ===
var hasNibeOnFinal = actions.indexOf("nibe_on") > -1;
var hasNibeOffFinal = actions.indexOf("nibe_off") > -1;
var hasPatOnFinal = actions.indexOf("p1_on") > -1 || actions.indexOf("p2_on") > -1 || actions.indexOf("p3_on") > -1;
var patronyRunning = p1on || p2on || p3on;
if (hasNibeOnFinal && (hasPatOnFinal || patronyRunning)) {
    actions = actions.filter(function(a) { return a !== "nibe_on"; });
    reason += " ⚠️MUTEX:NIBE blokováno patronami";
}
// v24: Pokud nibe_off je ve STEJNÉM batchi → patrony mohou startovat (NIBE se vypne v tomto cyklu)
if (hasPatOnFinal && nibeOn && !hasNibeOffFinal) {
    actions = actions.filter(function(a) { return a !== "p1_on" && a !== "p2_on" && a !== "p3_on"; });
    reason += " ⚠️MUTEX:patrony blokovány NIBE";
}
// v24: Zajistit správné pořadí — nibe_off PŘED p*_on (fyzická bezpečnost)
if (hasPatOnFinal && hasNibeOffFinal) {
    var nibeIdx = actions.indexOf("nibe_off");
    var patIdx = Math.min(
        actions.indexOf("p1_on") > -1 ? actions.indexOf("p1_on") : 999,
        actions.indexOf("p2_on") > -1 ? actions.indexOf("p2_on") : 999,
        actions.indexOf("p3_on") > -1 ? actions.indexOf("p3_on") : 999
    );
    if (nibeIdx > patIdx) {
        // Přesuň nibe_off na začátek
        actions = actions.filter(function(a) { return a !== "nibe_off"; });
        actions.unshift("nibe_off");
    }
}

// === TOPENÍ MOD: zapsat do global pro HA input_select ===
// topeniModAktivni je undefined pokud automatizace=OFF → použít manualMod nebo "Vypnuto"
var finalTopeniMod = (typeof topeniModAktivni !== "undefined" && topeniModAktivni) ? topeniModAktivni : (global.get("topeni_mod") || "Vypnuto");
global.set("topeni_mod", finalTopeniMod);

// === GLOBAL: cerpadlo_topi pro FVE blokaci ===
var isHeating = pumpState.indexOf("Vytápění") > -1;
var isTUV = pumpState.indexOf("Ohřev vody") > -1 || pumpState.indexOf("TUV") > -1;
// Pokud NIBE běží s velkým solárním ziskem, NEBLOKOVAT vybíjení baterie
var nibeBlkDisch = flow.get("nibe_block_discharge") !== false;  // default true
global.set("cerpadlo_topi", (isHeating || isTUV) && nibeBlkDisch);

// === STATUS ===
var st = [];
st.push(automatizace ? (letniRezim ? "LÉTO" : "ZIMA") : "OFF");
st.push(indoorTemp.toFixed(1) + "/" + effTarget.toFixed(1) + "°C");
st.push("Tank:" + tankTemp.toFixed(0));
st.push("Lv" + lvl);
if (actions.length > 0) st.push("→" + actions.join(","));

var color = "green";
if (!automatizace) color = "grey";
else if (indoorTemp < NOUZ_TEMP) color = "red";
else if (isDraha || patTarget > 0) color = "yellow";
node.status({fill: color, shape: "dot", text: st.join(" | ")});

// === OUTPUT ===
// O1: actions array → router switch
// O2: debug/status
var actionMsgs = null;
if (actions.length > 0) {
    actionMsgs = actions.map(function(a) { return {action: a}; });
}
var statusMsg = {
    payload: {
        automatizace: automatizace, letniRezim: letniRezim,
        indoorTemp: indoorTemp, targetTemp: effTarget,
        tankTemp: tankTemp, dm: dm, outdoorTemp: outdoorTemp,
        priceLevel: lvl, isDraha: isDraha, isNight: isNight,
        nibeOn: nibeOn, obehOn: obehOn, patrony: actPat, patTarget: patTarget,
        fireplace: fireplace, batSoc: batSoc, prebytek: prebytek,
        actions: actions, reason: reason
    }
};
return [actionMsgs, statusMsg];
